## Create Database
```sql
USE master;
GO
IF DB_ID (N'BLOODBANK') IS NOT NULL
DROP DATABASE BLOODBANK;
GO
CREATE DATABASE BLOODBANK;
GO
```
## Create "Donneurs" Table
```sql
USE [BLOODBANK]
GO

CREATE TABLE [dboCREATE TABLE dbo.DONNEURS(
	Id varchar(50) NOT NULL,
	NumCarte varchar(50) NOT NULL,
	Nom varchar(50) NOT NULL,
	Prenom varchar(50) NOT NULL,
	DDN datetime NOT NULL,
	LDN varchar(50) NOT NULL,
	Sexe int NOT NULL,
	Telephone varchar(50) NOT NULL,
	Email varchar(200) NOT NULL,
	Groupage int NOT NULL,
	Commune char(4) NULL,
 CONSTRAINT PK_DONNEURS PRIMARY KEY CLUSTERED (Id ASC)
 )

GO
```
