# DataBase
## Create Database
```sql
USE master;
GO
IF DB_ID (N'SCOLARITE') IS NOT NULL
DROP DATABASE SCOLARITE;
GO
CREATE DATABASE SCOLARITE;
GO
```
## Create "Etudiant" Table
```sql
USE [SCOLARITE]
GO

CREATE TABLE [dbo].[ETUDIANTS](
	[ID] [varchar](50) NOT NULL,
	[NumCarte] [varchar](50) NOT NULL,
	[Nom] [varchar](50) NOT NULL,
	[Prenom] [varchar](50) NOT NULL,
	[DDN] [datetime] NOT NULL,
	[LDN] [varchar](50) NOT NULL,
	[Email] [varchar](200) NOT NULL,
	[Telephone] [varchar](50) NOT NULL,
 	CONSTRAINT [PK_ETUDIANTS] PRIMARY KEY CLUSTERED ([Id] ASC)) 
GO
```
